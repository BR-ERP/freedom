processo para assinar e verificar o JAR
---
keytool -genkey -keystore keystore -alias myself
keytool -selfcert -keystore keystore -alias myself
keytool -list-keystore keystore
jarsigner -keystore keystore Freedom-1.1.0.2.jar myself
jarsigner -verify Freedom-1.1.0.2.jar

----
